1) if (!("a" in window)) { var a = 1; } alert(a); браузер выведет undefined потому что по условию if строка должна быть без значения, а она имеет числовое значение 1 2) var b = function a(x) { x && a(--x); }; alert(a); браузер не выведет ничего, т.к. "а" - это имя функции и не определено какое-либо действие. 3) function a(x) { return x * 2; } var a; alert(a); браузер не выведет ничего, т.к. переменная "a" без значения, а значит выводится функция "а", для которой значение переменной "x", указанной в функции, отсутствует. 4) function b(x, y, a) { arguments[2] = 10; alert(a); } b(1, 2, 3); В данном примере браузер выведет третий аргумент, которому присвоено значение 10 в теле функции. 5) function a() { alert(this); } a.call(null); Слово "this" обозначает объект, который запускает данную функцию. здесь его нет, значит "this" ничего не обозначает. Браузер понимает, что должен быть объект, но не видит какой. В "alert" будет указано, что это объект без названия.
